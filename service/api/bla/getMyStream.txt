package api

import (
	"WasaPhoto/service/api/reqcontext"
	"net/http"
	"strings"

	"github.com/julienschmidt/httprouter"
)

func (rt *_router) getMyStream(w http.ResponseWriter, r *http.Request, ps httprouter.Params, ctx reqcontext.RequestContext) {
	// Estrazione username dall'URL

	username := strings.Split(r.URL.Path, "/")[2]
	err, ex := rt.db.CheckExistence(username)
	if err != nil {
		w.WriteHeader(http.StatusInternalServerError)
		ctx.Logger.WithError(err).Error("Errore nel controllare l'esistenza dell'utente con username specificato.")
		return
	} else {
		if *ex == false {
			w.WriteHeader(http.StatusNotFound)
			return
		}
	}

}
